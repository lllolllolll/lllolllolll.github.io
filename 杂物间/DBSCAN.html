### 定义读取图片的函数和返回图片的函数
def read_page(path):
    # 读取webp文件
    img = Image.open(path)
    
    # 将图像转换为numpy数组
    img_array = np.array(img)
    
    return img_array

####返回为图片
def to_page(array):
    img = Image.fromarray(array)
    return img
